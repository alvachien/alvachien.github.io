# 逐行累加
```sql
SELECT ID, 
	SUM( NUM ) OVER( ORDER BY ID ) as sum_num
```

# 逐行相加
```sql
SELECT ID, NUM,
	NUM + coalesce(lag(num) over ( order by ID ),0) as result
```

```sql
SELECT ID, NUM,
	NUM - coalesce(lag(num) over ( order by ID ),0) as result
```
